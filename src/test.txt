    static int find_height(TreeNode root)
    {
        if(root == null)
            return -1;

        if(root.children.size() == 0)
            return 0;

        int h = Integer.MIN_VALUE;
        for(TreeNode child : root.children) {
            h = Math.max(h, find_height(child));
        }
        return 1 + h;
    }

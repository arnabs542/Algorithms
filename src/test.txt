    static String[] find_all_well_formed_brackets(int n) {
        List<String> l = new ArrayList<>();

        helper(0, 0, "", l, n);

        return l.toArray(new String[0]);
    }

    static void helper(int open, int close, String sofar, List<String> l, int n) {
        if(close > open || close > n || open > n) {
            return;
        }
        if(close == n && open == n) {
            l.add(sofar);
        } else {
            helper(open+1, close, sofar+"(", l, n);
            helper(open, close+1, sofar+")", l, n);
        }
    }
